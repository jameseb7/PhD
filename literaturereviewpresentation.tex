\section{Motivation}

\begin{slide}{Motivation}
\begin{itemize}
\item Popularity of Java\cite{gosling2013}
\item Variants of Java: JavaCard\cite{chen2000}, RTSJ\cite{gosling2000}, SCJ\cite{locke2013}
\item Program correctness relies on compiler/interpreter correctness.
\item Testing is usually not sufficient to ensure correctness.
\end{itemize}
\end{slide}

\section{Compiler Correctness}

\frame{\sectionpage}

\subsection{Early Work}

\begin{slide}{Early Work}
  \begin{itemize}
  \item McCarthy and Painter\cite{mccarthy1967}
    \begin{itemize}
    \item Shows correctness of a compiler for a simple expression language
    \item Targets a simple single-register machine
    \item Correctness defined in terms of partial equality of machine states
    \end{itemize}
  \item Burstall and Landin\cite{burstall1969}
    \begin{itemize}
    \item More algebraic treatment of a compiler for an expression language
    \item Targets several intermediate machines including a stack machine and a machine similar to McCarthy and Painter's
    \item Shows correctness by constructing homomorphisms between algebras
    \end{itemize}
  \end{itemize}
\end{slide}

\begin{slide}{Early Work}
  \begin{itemize}
  \item Milner and Weyhrauch\cite{milner1972}
    \begin{itemize}
    \item Partially mechanised proof of compiler correctness for a simple ALGOL-like language
    \item Targets a simple stack machine that allows jumps
    \item Defines correctness as an equality between the source semantics and the composition of the compilation function, the target semantics and a function to extract the source state of the machine
    \end{itemize}
  \end{itemize}
\end{slide}

\subsection{Agebraic Approach}

\begin{slide}{Agebraic Approach}
  \begin{itemize}
  \item Proposed by Hoare in 1991\cite{hoare1991}
  \item Both the source and target languages are described using the \emph{same specification language} with laws for reasoning about them \cite{hoare1987}.
  \item Uses the concept of a \emph{refinement relation} between programs: $P \sqsubseteq Q$ means the program $Q$ is at least as good as $P$ \cite{back1981, morris1987, morgan1990}.
  \item Reduces source program to a \emph{normal form} that resembles an interpreter for the target machine.
  \end{itemize}
\end{slide}

\begin{slide}{Agebraic Approach}
  \begin{itemize}
  \item Sampaio\cite{hoare1993, sampaio1993, sampaio1997}
    \begin{itemize}
    \item Handles most imperative constructs, including procedures and recursion.
    \item Performs compilation  using rewrite rules proved from basic laws.
    \item Mechanises compilation in the OBJ3 term rewriting system\cite{goguen1988}.
    \end{itemize}
  \item Perna\cite{perna2010, perna2011} --- Hardware compilation
    \begin{itemize}
    \item Handles timed structures and parallelism with shared variables
    \end{itemize}
    % \item \cite{hoare1984}\cite{morgan1990}
  \end{itemize}
\end{slide}

\subsection{Recent Work}

\begin{slide}{Recent Work}
  \begin{itemize}
  \item CompCert\cite{leroy2012}
    \begin{itemize}
    \item Project to develop a realistic formally verified compiler
    \item Produced a formally verified compiler for a large subset of C
    \item Compiler developed and proved correct in the Coq proof assistant
    \end{itemize}
  \item Wang, Cuellar and Chipala\cite{wang2014} --- Verifying compilers that allow linking with other languages
    \begin{itemize}
    \item Provides a combined algebraic and operational semantics of the source language
    \item Allows for algebraic reasoning about calls to programs in other languages
    \item Mechanised in Coq 
    \end{itemize}
  \end{itemize}
\end{slide}

\section{Compiler Correctness for Java-like Languages}

\frame{\sectionpage}
\stepcounter{subsection}

\begin{slide}{ASM approach}
St\"{a}rk, Schmid and B\"orger\cite{stark2001}
\begin{itemize}
\item Defines Java and the JVM in terms of abstract state machines (ASMs)
\item Splits Java into sublanguages: imperative, procedural, object-oriented, exception-handling, concurrent
\item Requires equivalences between the Java ASM and the JVM ASM to be satisfied for the compilation to be correct
\item Proves correctness by induction over the structure of Java code
\end{itemize}
\end{slide}

\begin{slide}{ROOL}
%borba2000, borba2004, cavalcanti2000
ROOL (Refinement Object-Oriented Language)\cite{cavalcanti2000}
\begin{itemize}
\item Subset of Java with specification features
\item Verified compiler using the normal form approach\cite{duran2005, duran2010}
  \begin{itemize}
  \item Adds class precompilation and redirection of method calls to
Sampaio’s phases of compilation
  \item Normal form representing a VM
  \end{itemize}
\end{itemize}
\end{slide}

\begin{slide}{Java Compilers in Isabelle/HOL}
  \begin{itemize}
  \item Strecker\cite{strecker2002}
    \begin{itemize}
    \item Develops a compiler for $\mu$Java, a subset of Java that contains many core features of Java but removes interfaces, arrays, access modifiers and concurrency
    \item Correctness is shown via a ``commuting diagram'' argument
    \end{itemize}
  \item Klein and Nipkow\cite{klein2006}
    \begin{itemize}
    \item Proves correctness of a compiler and JVM for a slightly larger subset of Java called Jinja
    \end{itemize}
  \item Lochbihler\cite{lochbihler2010}
    \begin{itemize}
    \item Adds support for concurrency to the verified compiler presented by Klein and Nipkow
    \end{itemize}
  \end{itemize}
\end{slide}

\section{Conclusion}

\begin{frame}
\frametitle{Conclusion}
  \begin{itemize}
  \item There are two main approaches to compilation
    \begin{itemize}
    \item The aglebraic approach offer advantages over the more traditional approach
    \end{itemize}
  \item Work has been done on verifying compilers for various languages, including Java
  \item There seems to be little work on verifying Java compilers for embedded systems
  \end{itemize}
\end{frame}
